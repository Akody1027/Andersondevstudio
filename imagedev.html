document.getElementById('btnConfirmCrop').addEventListener('click', () => { if (!shapePath || !activeImgToCrop) return; // 1. Calculate relative center point (Canvas Space -> Image Space) const imgMatrix = activeImgToCrop.calcTransformMatrix(); const invertedImgMatrix = fabric.util.invertTransform(imgMatrix); const shapeCenter = shapePath.getCenterPoint(); const relativePoint = fabric.util.transformPoint(shapeCenter, invertedImgMatrix); // 2. Remove the orange viewfinder from the canvas UI canvas.remove(shapePath); // 3. Convert the viewfinder itself into a relative clipping mask shapePath.set({ left: relativePoint.x, top: relativePoint.y, scaleX: shapePath.scaleX / activeImgToCrop.scaleX, scaleY: shapePath.scaleY / activeImgToCrop.scaleY, angle: shapePath.angle - (activeImgToCrop.angle || 0), originX: 'center', originY: 'center', fill: '#000000', strokeWidth: 0, absolutePositioned: false }); // 4. Apply mask and re-render activeImgToCrop.set('clipPath', shapePath); shapePath = null; canvas.setActiveObject(activeImgToCrop); closeAllPanels(); canvas.renderAll(); });
